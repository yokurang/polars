19/// Read the number of rows without parsing columns
20/// useful for count(*) queries
21pub fn count_rows(
22    path: &Path,
23    separator: u8,
24    quote_char: Option<u8>,
25    comment_prefix: Option<&CommentPrefix>,
26    eol_char: u8,
27    has_header: bool,
28) -> PolarsResult<usize> {
29    let file = if is_cloud_url(path) || config::force_async() {
30        feature_gated!("cloud", {
31            crate::file_cache::FILE_CACHE
32                .get_entry(path.to_str().unwrap())
33                // Safety: This was initialized by schema inference.
34                .unwrap()
35                .try_open_assume_latest()?
36        })
37    } else {
38        polars_utils::open_file(path)?
39    };
40
41    let mmap = unsafe { memmap::Mmap::map(&file).unwrap() };
42    let owned = &mut vec![];
43    let reader_bytes = maybe_decompress_bytes(mmap.as_ref(), owned)?;
44
45    count_rows_from_slice_par(
46        reader_bytes,
47        separator,
48        quote_char,
49        comment_prefix,
50        eol_char,
51        has_header,
52    )
53}

55/// Read the number of rows without parsing columns
56/// useful for count(*) queries
57pub fn count_rows_from_slice_par(
58    mut bytes: &[u8],
59    separator: u8,
60    quote_char: Option<u8>,
61    comment_prefix: Option<&CommentPrefix>,
62    eol_char: u8,
63    has_header: bool,
64) -> PolarsResult<usize> {
65    for _ in 0..bytes.len() {
66        if bytes[0] != eol_char {
67            break;
68        }
69
70        bytes = &bytes[1..];
71    }
72
73    const MIN_ROWS_PER_THREAD: usize = 1024;
74    let max_threads = POOL.current_num_threads();
75
76    // Determine if parallelism is beneficial and how many threads
77    let n_threads = get_line_stats(
78        bytes,
79        MIN_ROWS_PER_THREAD,
80        eol_char,
81        None,
82        separator,
83        quote_char,
84    )
85    .map(|(mean, std)| {
86        let n_rows = (bytes.len() as f32 / (mean - 0.01 * std)) as usize;
87        (n_rows / MIN_ROWS_PER_THREAD).clamp(1, max_threads)
88    })
89    .unwrap_or(1);
90
91    if n_threads == 1 {
92        return count_rows_from_slice(bytes, quote_char, comment_prefix, eol_char, has_header);
93    }
94
95    let file_chunks: Vec<(usize, usize)> =
96        get_file_chunks(bytes, n_threads, None, separator, quote_char, eol_char);
97
98    let iter = file_chunks.into_par_iter().map(|(start, stop)| {
99        let bytes = &bytes[start..stop];
100
101        if comment_prefix.is_some() {
102            SplitLines::new(bytes, quote_char, eol_char, comment_prefix)
103                .filter(|line| !is_comment_line(line, comment_prefix))
104                .count()
105        } else {
106            CountLines::new(quote_char, eol_char).count(bytes).0
107        }
108    });
109
110    let n: usize = POOL.install(|| iter.sum());
111
112    Ok(n - (has_header as usize))
113}

152/// Checks if a line in a CSV file is a comment based on the given comment prefix configuration.
153///
154/// This function is used during CSV parsing to determine whether a line should be ignored based on its starting characters.
155#[inline]
156pub(super) fn is_comment_line(line: &[u8], comment_prefix: Option<&CommentPrefix>) -> bool {
157    match comment_prefix {
158        Some(CommentPrefix::Single(c)) => line.first() == Some(c),
159        Some(CommentPrefix::Multi(s)) => line.starts_with(s.as_bytes()),
160        None => false,
161    }
162}

164/// Find the nearest next line position.
165/// Does not check for new line characters embedded in String fields.
166pub(super) fn next_line_position_naive(input: &[u8], eol_char: u8) -> Option<usize> {
167    let pos = memchr::memchr(eol_char, input)? + 1;
168    if input.len() - pos == 0 {
169        return None;
170    }
171    Some(pos)
172}

174pub(super) fn skip_lines_naive(mut input: &[u8], eol_char: u8, skip: usize) -> &[u8] {
175    for _ in 0..skip {
176        if let Some(pos) = next_line_position_naive(input, eol_char) {
177            input = &input[pos..];
178        } else {
179            return input;
180        }
181    }
182    input
183}

185/// Find the nearest next line position that is not embedded in a String field.
186pub(super) fn next_line_position(
187    mut input: &[u8],
188    mut expected_fields: Option<usize>,
189    separator: u8,
190    quote_char: Option<u8>,
191    eol_char: u8,
192) -> Option<usize> {
193    fn accept_line(
194        line: &[u8],
195        expected_fields: usize,
196        separator: u8,
197        eol_char: u8,
198        quote_char: Option<u8>,
199    ) -> bool {
200        let mut count = 0usize;
201        for (field, _) in SplitFields::new(line, separator, quote_char, eol_char) {
202            if memchr2_iter(separator, eol_char, field).count() >= expected_fields {
203                return false;
204            }
205            count += 1;
206        }
207
208        // if the latest field is missing
209        // e.g.:
210        // a,b,c
211        // vala,valb,
212        // SplitFields returns a count that is 1 less
213        // There fore we accept:
214        // expected == count
215        // and
216        // expected == count - 1
217        expected_fields.wrapping_sub(count) <= 1
218    }
219
220    // we check 3 subsequent lines for `accept_line` before we accept
221    // if 3 groups are rejected we reject completely
222    let mut rejected_line_groups = 0u8;
223
224    let mut total_pos = 0;
225    if input.is_empty() {
226        return None;
227    }
228    let mut lines_checked = 0u8;
229    loop {
230        if rejected_line_groups >= 3 {
231            return None;
232        }
233        lines_checked = lines_checked.wrapping_add(1);
234        // headers might have an extra value
235        // So if we have churned through enough lines
236        // we try one field less.
237        if lines_checked == u8::MAX {
238            if let Some(ef) = expected_fields {
239                expected_fields = Some(ef.saturating_sub(1))
240            }
241        };
242        let pos = memchr::memchr(eol_char, input)? + 1;
243        if input.len() - pos == 0 {
244            return None;
245        }
246        debug_assert!(pos <= input.len());
247        let new_input = unsafe { input.get_unchecked(pos..) };
248        let mut lines = SplitLines::new(new_input, quote_char, eol_char, None);
249        let line = lines.next();
250
251        match (line, expected_fields) {
252            // count the fields, and determine if they are equal to what we expect from the schema
253            (Some(line), Some(expected_fields)) => {
254                if accept_line(line, expected_fields, separator, eol_char, quote_char) {
255                    let mut valid = true;
256                    for line in lines.take(2) {
257                        if !accept_line(line, expected_fields, separator, eol_char, quote_char) {
258                            valid = false;
259                            break;
260                        }
261                    }
262                    if valid {
263                        return Some(total_pos + pos);
264                    } else {
265                        rejected_line_groups += 1;
266                    }
267                } else {
268                    debug_assert!(pos < input.len());
269                    unsafe {
270                        input = input.get_unchecked(pos + 1..);
271                    }
272                    total_pos += pos + 1;
273                }
274            },
275            // don't count the fields
276            (Some(_), None) => return Some(total_pos + pos),
277            // // no new line found, check latest line (without eol) for number of fields
278            _ => return None,
279        }
280    }
281}

291#[inline]
292fn skip_condition<F>(input: &[u8], f: F) -> &[u8]
293where
294    F: Fn(u8) -> bool,
295{
296    if input.is_empty() {
297        return input;
298    }
299
300    let read = input.iter().position(|b| !f(*b)).unwrap_or(input.len());
301    &input[read..]
302}

319/// Get the mean and standard deviation of length of lines in bytes
320pub(super) fn get_line_stats(
321    bytes: &[u8],
322    n_lines: usize,
323    eol_char: u8,
324    expected_fields: Option<usize>,
325    separator: u8,
326    quote_char: Option<u8>,
327) -> Option<(f32, f32)> {
328    let mut lengths = Vec::with_capacity(n_lines);
329
330    let mut bytes_trunc;
331    let n_lines_per_iter = n_lines / 2;
332
333    let mut n_read = 0;
334
335    // sample from start and 75% in the file
336    for offset in [0, (bytes.len() as f32 * 0.75) as usize] {
337        bytes_trunc = &bytes[offset..];
338        let pos = next_line_position(
339            bytes_trunc,
340            expected_fields,
341            separator,
342            quote_char,
343            eol_char,
344        )?;
345        bytes_trunc = &bytes_trunc[pos + 1..];
346
347        for _ in offset..(offset + n_lines_per_iter) {
348            let pos = next_line_position_naive(bytes_trunc, eol_char)? + 1;
349            n_read += pos;
350            lengths.push(pos);
351            bytes_trunc = &bytes_trunc[pos..];
352        }
353    }
354
355    let n_samples = lengths.len();
356
357    let mean = (n_read as f32) / (n_samples as f32);
358    let mut std = 0.0;
359    for &len in lengths.iter() {
360        std += (len as f32 - mean).pow(2.0)
361    }
362    std = (std / n_samples as f32).sqrt();
363    Some((mean, std))
364}

366/// An adapted version of std::iter::Split.
367/// This exists solely because we cannot split the file in lines naively as
368///
369/// ```text
370///    for line in bytes.split(b'\n') {
371/// ```
372///
373/// This will fail when strings fields are have embedded end line characters.
374/// For instance: "This is a valid field\nI have multiples lines" is a valid string field, that contains multiple lines.
375pub(super) struct SplitLines<'a> {
376    v: &'a [u8],
377    quote_char: u8,
378    eol_char: u8,
379    #[cfg(feature = "simd")]
380    simd_eol_char: SimdVec,
381    #[cfg(feature = "simd")]
382    simd_quote_char: SimdVec,
383    #[cfg(feature = "simd")]
384    previous_valid_eols: u64,
385    total_index: usize,
386    quoting: bool,
387    comment_prefix: Option<&'a CommentPrefix>,
388}

390#[cfg(feature = "simd")]
391const SIMD_SIZE: usize = 64;
392#[cfg(feature = "simd")]
393use std::simd::prelude::*;

395#[cfg(feature = "simd")]
396use polars_utils::clmul::prefix_xorsum_inclusive;

398#[cfg(feature = "simd")]

401impl<'a> SplitLines<'a> {
402    pub(super) fn new(

409        let quote_char = quote_char.unwrap_or(b'\"');

412        #[cfg(feature = "simd")]
413        let simd_quote_char = SimdVec::splat(quote_char);
414        Self {
415            v: slice,
416            quote_char,
417            eol_char,
418            #[cfg(feature = "simd")]
419            simd_eol_char,
420            #[cfg(feature = "simd")]
421            simd_quote_char,
422            #[cfg(feature = "simd")]
423            previous_valid_eols: 0,
424            total_index: 0,
425            quoting,
426            comment_prefix,
427        }
428    }

431impl<'a> SplitLines<'a> {
432    // scalar as in non-simd
433    fn next_scalar(&mut self) -> Option<&'a [u8]> {
434        if self.v.is_empty() {
435            return None;
436        }
437        if is_comment_line(self.v, self.comment_prefix) {
438            return self.next_comment_line();
439        }
440        {
441            let mut pos = 0u32;
442            let mut iter = self.v.iter();
443            let mut in_field = false;
444            loop {
445                match iter.next() {
446                    Some(&c) => {
447                        pos += 1;
448
449                        if self.quoting && c == self.quote_char {
450                            // toggle between string field enclosure
451                            //      if we encounter a starting '"' -> in_field = true;
452                            //      if we encounter a closing '"' -> in_field = false;
453                            in_field = !in_field;
454                        }
455                        // if we are not in a string and we encounter '\n' we can stop at this position.
456                        else if c == self.eol_char && !in_field {
457                            break;
458                        }
459                    },
460                    None => {
461                        let remainder = self.v;
462                        self.v = &[];
463                        return Some(remainder);
464                    },
465                }
466            }
467
468            unsafe {
469                debug_assert!((pos as usize) <= self.v.len());
470
471                // return line up to this position
472                let ret = Some(
473                    self.v
474                        .get_unchecked(..(self.total_index + pos as usize - 1)),
475                );
476                // skip the '\n' token and update slice.
477                self.v = self.v.get_unchecked(self.total_index + pos as usize..);
478                ret
479            }
480        }
481    }
482    fn next_comment_line(&mut self) -> Option<&'a [u8]> {
483        if let Some(pos) = next_line_position_naive(self.v, self.eol_char) {
484            unsafe {
485                // return line up to this position
486                let ret = Some(self.v.get_unchecked(..(pos - 1)));
487                // skip the '\n' token and update slice.
488                self.v = self.v.get_unchecked(pos..);
489                ret
490            }
491        } else {
492            let remainder = self.v;
493            self.v = &[];
494            Some(remainder)
495        }
496    }
497}

499impl<'a> Iterator for SplitLines<'a> {
500    type Item = &'a [u8];
501
502    #[inline]
503    #[cfg(not(feature = "simd"))]
504    fn next(&mut self) -> Option<&'a [u8]> {
505        self.next_scalar()
506    }
507
508    #[inline]
509    #[cfg(feature = "simd")]
510    fn next(&mut self) -> Option<&'a [u8]> {
511        // First check cached value
512        if self.previous_valid_eols != 0 {
513            let pos = self.previous_valid_eols.trailing_zeros() as usize;
514            self.previous_valid_eols >>= (pos + 1) as u64;
515
516            unsafe {
517                debug_assert!((pos) <= self.v.len());
518
519                // return line up to this position
520                let ret = Some(self.v.get_unchecked(..pos));
521                // skip the '\n' token and update slice.
522                self.v = self.v.get_unchecked(pos + 1..);
523                return ret;
524            }
525        }
526        if self.v.is_empty() {
527            return None;
528        }
529        if self.comment_prefix.is_some() {
530            return self.next_scalar();
531        }
532
533        self.total_index = 0;
534        let mut not_in_field_previous_iter = true;
535
536        loop {
537            let bytes = unsafe { self.v.get_unchecked(self.total_index..) };
538            if bytes.len() > SIMD_SIZE {
539                let lane: [u8; SIMD_SIZE] = unsafe {
540                    bytes
541                        .get_unchecked(0..SIMD_SIZE)
542                        .try_into()
543                        .unwrap_unchecked()
544                };
545                let simd_bytes = SimdVec::from(lane);
546                let eol_mask = simd_bytes.simd_eq(self.simd_eol_char).to_bitmask();
547
548                let valid_eols = if self.quoting {
549                    let quote_mask = simd_bytes.simd_eq(self.simd_quote_char).to_bitmask();
550                    let mut not_in_quote_field = prefix_xorsum_inclusive(quote_mask);
551
552                    if not_in_field_previous_iter {
553                        not_in_quote_field = !not_in_quote_field;
554                    }
555                    not_in_field_previous_iter = (not_in_quote_field & (1 << (SIMD_SIZE - 1))) > 0;
556                    eol_mask & not_in_quote_field
557                } else {
558                    eol_mask
559                };
560
561                if valid_eols != 0 {
562                    let pos = valid_eols.trailing_zeros() as usize;
563                    if pos == SIMD_SIZE - 1 {
564                        self.previous_valid_eols = 0;
565                    } else {
566                        self.previous_valid_eols = valid_eols >> (pos + 1) as u64;
567                    }
568
569                    unsafe {
570                        let pos = self.total_index + pos;
571                        debug_assert!((pos) <= self.v.len());
572
573                        // return line up to this position
574                        let ret = Some(self.v.get_unchecked(..pos));
575                        // skip the '\n' token and update slice.
576                        self.v = self.v.get_unchecked(pos + 1..);
577                        return ret;
578                    }
579                } else {
580                    self.total_index += SIMD_SIZE;
581                }
582            } else {
583                // Denotes if we are in a string field, started with a quote
584                let mut in_field = !not_in_field_previous_iter;
585                let mut pos = 0u32;
586                let mut iter = bytes.iter();
587                loop {
588                    match iter.next() {
589                        Some(&c) => {
590                            pos += 1;
591
592                            if self.quoting && c == self.quote_char {
593                                // toggle between string field enclosure
594                                //      if we encounter a starting '"' -> in_field = true;
595                                //      if we encounter a closing '"' -> in_field = false;
596                                in_field = !in_field;
597                            }
598                            // if we are not in a string and we encounter '\n' we can stop at this position.
599                            else if c == self.eol_char && !in_field {
600                                break;
601                            }
602                        },
603                        None => {
604                            let remainder = self.v;
605                            self.v = &[];
606                            return Some(remainder);
607                        },
608                    }
609                }
610
611                unsafe {
612                    debug_assert!((pos as usize) <= self.v.len());
613
614                    // return line up to this position
615                    let ret = Some(
616                        self.v
617                            .get_unchecked(..(self.total_index + pos as usize - 1)),
618                    );
619                    // skip the '\n' token and update slice.
620                    self.v = self.v.get_unchecked(self.total_index + pos as usize..);
621                    return ret;
622                }
623            }
624        }
625    }
626}

638#[derive(Copy, Clone, Debug)]
639pub struct LineStats {
640    newline_count: usize,
641    last_newline_offset: usize,
642    end_inside_string: bool,
643}
644
645impl CountLines {
646    pub fn new(quote_char: Option<u8>, eol_char: u8) -> Self {
647        let quoting = quote_char.is_some();
648        let quote_char = quote_char.unwrap_or(b'\"');
649        #[cfg(feature = "simd")]
650        let simd_eol_char = SimdVec::splat(eol_char);
651        #[cfg(feature = "simd")]
652        let simd_quote_char = SimdVec::splat(quote_char);
653        Self {
654            quote_char,
655            eol_char,
656            #[cfg(feature = "simd")]
657            simd_eol_char,
658            #[cfg(feature = "simd")]
659            simd_quote_char,
660            quoting,
661        }
662    }
663
664    /// Analyzes a chunk of CSV data.
665    ///
666    /// Returns (newline_count, last_newline_offset, end_inside_string) twice,
667    /// the first is assuming the start of the chunk is *not* inside a string,
668    /// the second assuming the start is inside a string.
669    pub fn analyze_chunk(&self, bytes: &[u8]) -> [LineStats; 2] {
670        let mut scan_offset = 0;
671        let mut states = [
672            LineStats {
673                newline_count: 0,
674                last_newline_offset: 0,
675                end_inside_string: false,
676            },
677            LineStats {
678                newline_count: 0,
679                last_newline_offset: 0,
680                end_inside_string: false,
681            },
682        ];
683
684        // false if even number of quotes seen so far, true otherwise.
685        #[allow(unused_assignments)]
686        let mut global_quote_parity = false;
687
688        #[cfg(feature = "simd")]
689        {
690            // 0 if even number of quotes seen so far, u64::MAX otherwise.
691            let mut global_quote_parity_mask = 0;
692            while scan_offset + 64 <= bytes.len() {
693                let block: [u8; 64] = unsafe {
694                    bytes
695                        .get_unchecked(scan_offset..scan_offset + 64)
696                        .try_into()
697                        .unwrap_unchecked()
698                };
699                let simd_bytes = SimdVec::from(block);
700                let eol_mask = simd_bytes.simd_eq(self.simd_eol_char).to_bitmask();
701                if self.quoting {
702                    let quote_mask = simd_bytes.simd_eq(self.simd_quote_char).to_bitmask();
703                    let quote_parity =
704                        prefix_xorsum_inclusive(quote_mask) ^ global_quote_parity_mask;
705                    global_quote_parity_mask = ((quote_parity as i64) >> 63) as u64;
706
707                    let start_outside_string_eol_mask = eol_mask & !quote_parity;
708                    states[0].newline_count += start_outside_string_eol_mask.count_ones() as usize;
709                    states[0].last_newline_offset = select_unpredictable(
710                        start_outside_string_eol_mask != 0,
711                        (scan_offset + 63)
712                            .wrapping_sub(start_outside_string_eol_mask.leading_zeros() as usize),
713                        states[0].last_newline_offset,
714                    );
715
716                    let start_inside_string_eol_mask = eol_mask & quote_parity;
717                    states[1].newline_count += start_inside_string_eol_mask.count_ones() as usize;
718                    states[1].last_newline_offset = select_unpredictable(
719                        start_inside_string_eol_mask != 0,
720                        (scan_offset + 63)
721                            .wrapping_sub(start_inside_string_eol_mask.leading_zeros() as usize),
722                        states[1].last_newline_offset,
723                    );
724                } else {
725                    states[0].newline_count += eol_mask.count_ones() as usize;
726                    states[0].last_newline_offset = select_unpredictable(
727                        eol_mask != 0,
728                        (scan_offset + 63).wrapping_sub(eol_mask.leading_zeros() as usize),
729                        states[0].last_newline_offset,
730                    );
731                }
732
733                scan_offset += 64;
734            }
735
736            global_quote_parity = global_quote_parity_mask > 0;
737        }
738
739        while scan_offset < bytes.len() {
740            let c = unsafe { *bytes.get_unchecked(scan_offset) };
741            global_quote_parity ^= (c == self.quote_char) & self.quoting;
742
743            let state = &mut states[global_quote_parity as usize];
744            state.newline_count += (c == self.eol_char) as usize;
745            state.last_newline_offset =
746                select_unpredictable(c == self.eol_char, scan_offset, state.last_newline_offset);
747
748            scan_offset += 1;
749        }
750
751        states[0].end_inside_string = global_quote_parity;
752        states[1].end_inside_string = !global_quote_parity;
753        states
754    }

756    pub fn find_next(&self, bytes: &[u8], chunk_size: &mut usize) -> (usize, usize) {
757        loop {
758            let b = unsafe { bytes.get_unchecked(..(*chunk_size).min(bytes.len())) };
759
760            let (count, offset) = self.count(b);
761
762            if count > 0 || b.len() == bytes.len() {
763                return (count, offset);
764            }
765
766            *chunk_size *= 2;
767        }
768    }

770    /// Returns count and offset to split for remainder in slice.
771    #[cfg(feature = "simd")]
772    pub fn count(&self, bytes: &[u8]) -> (usize, usize) {
773        let mut total_idx = 0;
774        let original_bytes = bytes;
775        let mut count = 0;
776        let mut position = 0;
777        let mut not_in_field_previous_iter = true;
778
779        loop {
780            let bytes = unsafe { original_bytes.get_unchecked(total_idx..) };
781
782            if bytes.len() > SIMD_SIZE {
783                let lane: [u8; SIMD_SIZE] = unsafe {
784                    bytes
785                        .get_unchecked(0..SIMD_SIZE)
786                        .try_into()
787                        .unwrap_unchecked()
788                };
789                let simd_bytes = SimdVec::from(lane);
790                let eol_mask = simd_bytes.simd_eq(self.simd_eol_char).to_bitmask();
791
792                let valid_eols = if self.quoting {
793                    let quote_mask = simd_bytes.simd_eq(self.simd_quote_char).to_bitmask();
794                    let mut not_in_quote_field = prefix_xorsum_inclusive(quote_mask);
795
796                    if not_in_field_previous_iter {
797                        not_in_quote_field = !not_in_quote_field;
798                    }
799                    not_in_field_previous_iter = (not_in_quote_field & (1 << (SIMD_SIZE - 1))) > 0;
800                    eol_mask & not_in_quote_field
801                } else {
802                    eol_mask
803                };
804
805                if valid_eols != 0 {
806                    count += valid_eols.count_ones() as usize;
807                    position = total_idx + 63 - valid_eols.leading_zeros() as usize;
808                    debug_assert_eq!(original_bytes[position], self.eol_char)
809                }
810                total_idx += SIMD_SIZE;
811            } else if bytes.is_empty() {
812                debug_assert!(count == 0 || original_bytes[position] == self.eol_char);
813                return (count, position);
814            } else {
815                let (c, o) = self.count_no_simd(bytes, !not_in_field_previous_iter);
816
817                let (count, position) = if c > 0 {
818                    (count + c, total_idx + o)
819                } else {
820                    (count, position)
821                };
822                debug_assert!(count == 0 || original_bytes[position] == self.eol_char);
823
824                return (count, position);
825            }
826        }
827    }

829    #[cfg(not(feature = "simd"))]
830    pub fn count(&self, bytes: &[u8]) -> (usize, usize) {
831        self.count_no_simd(bytes, false)
832    }

834    fn count_no_simd(&self, bytes: &[u8], in_field: bool) -> (usize, usize) {
835        let iter = bytes.iter();
836        let mut in_field = in_field;
837        let mut count = 0;
838        let mut position = 0;
839
840        for b in iter {
841            let c = *b;
842            if self.quoting && c == self.quote_char {
843                // toggle between string field enclosure
844                //      if we encounter a starting '"' -> in_field = true;
845                //      if we encounter a closing '"' -> in_field = false;
846                in_field = !in_field;
847            }
848            // If we are not in a string and we encounter '\n' we can stop at this position.
849            else if c == self.eol_char && !in_field {
850                position = (b as *const _ as usize) - (bytes.as_ptr() as usize);
851                count += 1;
852            }
853        }
854        debug_assert!(count == 0 || bytes[position] == self.eol_char);
855
856        (count, position)
857    }

860#[inline]
861fn find_quoted(bytes: &[u8], quote_char: u8, needle: u8) -> Option<usize> {
862    let mut in_field = false;
863
864    let mut idx = 0u32;
865    // micro optimizations
866    #[allow(clippy::explicit_counter_loop)]
867    for &c in bytes.iter() {
868        if c == quote_char {
869            // toggle between string field enclosure
870            //      if we encounter a starting '"' -> in_field = true;
871            //      if we encounter a closing '"' -> in_field = false;
872            in_field = !in_field;
873        }
874
875        if !in_field && c == needle {
876            return Some(idx as usize);
877        }
878        idx += 1;
879    }
880    None
881}

883#[inline]
884pub(super) fn skip_this_line(bytes: &[u8], quote: Option<u8>, eol_char: u8) -> &[u8] {
885    let pos = match quote {
886        Some(quote) => find_quoted(bytes, quote, eol_char),
887        None => bytes.iter().position(|x| *x == eol_char),
888    };
889    match pos {
890        None => &[],
891        Some(pos) => &bytes[pos + 1..],
892    }
893}

895#[inline]
896pub(super) fn skip_this_line_naive(input: &[u8], eol_char: u8) -> &[u8] {
897    if let Some(pos) = next_line_position_naive(input, eol_char) {
898        unsafe { input.get_unchecked(pos..) }
899    } else {
900        &[]
901    }

904/// Parse CSV.
905///
906/// # Arguments
907/// * `bytes` - input to parse
908/// * `offset` - offset in bytes in total input. This is 0 if single threaded. If multi-threaded every
909///   thread has a different offset.
910/// * `projection` - Indices of the columns to project.
911/// * `buffers` - Parsed output will be written to these buffers. Except for UTF8 data. The offsets of the
912///   fields are written to the buffers. The UTF8 data will be parsed later.
913#[allow(clippy::too_many_arguments)]
914pub(super) fn parse_lines(
915    mut bytes: &[u8],
916    parse_options: &CsvParseOptions,
917    offset: usize,
918    ignore_errors: bool,
919    null_values: Option<&NullValuesCompiled>,
920    projection: &[usize],
921    buffers: &mut [Buffer],
922    n_lines: usize,
923    // length of original schema
924    schema_len: usize,
925    schema: &Schema,
926) -> PolarsResult<usize> {
927    assert!(
928        !projection.is_empty(),
929        "at least one column should be projected"
930    );
931    let mut truncate_ragged_lines = parse_options.truncate_ragged_lines;
932    // During projection pushdown we are not checking other csv fields.
933    // This would be very expensive and we don't care as we only want
934    // the projected columns.
935    if projection.len() != schema_len {
936        truncate_ragged_lines = true
937    }
938
939    // we use the pointers to track the no of bytes read.
940    let start = bytes.as_ptr() as usize;
941    let original_bytes_len = bytes.len();
942    let n_lines = n_lines as u32;
943
944    let mut line_count = 0u32;
945    loop {
946        if line_count > n_lines {
947            let end = bytes.as_ptr() as usize;
948            return Ok(end - start);
949        }
950
951        if bytes.is_empty() {
952            return Ok(original_bytes_len);
953        } else if is_comment_line(bytes, parse_options.comment_prefix.as_ref()) {
954            // deal with comments
955            let bytes_rem = skip_this_line_naive(bytes, parse_options.eol_char);
956            bytes = bytes_rem;
957            continue;
958        }
959
960        // Every line we only need to parse the columns that are projected.
961        // Therefore we check if the idx of the field is in our projected columns.
962        // If it is not, we skip the field.
963        let mut projection_iter = projection.iter().copied();
964        let mut next_projected = unsafe { projection_iter.next().unwrap_unchecked() };
965        let mut processed_fields = 0;
966
967        let mut iter = SplitFields::new(
968            bytes,
969            parse_options.separator,
970            parse_options.quote_char,
971            parse_options.eol_char,
972        );
973        let mut idx = 0u32;
974        let mut read_sol = 0;
975        loop {
976            match iter.next() {
977                // end of line
978                None => {
979                    bytes = unsafe { bytes.get_unchecked(std::cmp::min(read_sol, bytes.len())..) };
980                    break;
981                },
982                Some((mut field, needs_escaping)) => {
983                    let field_len = field.len();
984
985                    // +1 is the split character that is consumed by the iterator.
986                    read_sol += field_len + 1;
987
988                    if idx == next_projected as u32 {
989                        // the iterator is finished when it encounters a `\n`
990                        // this could be preceded by a '\r'
991                        unsafe {
992                            if field_len > 0 && *field.get_unchecked(field_len - 1) == b'\r' {
993                                field = field.get_unchecked(..field_len - 1);
994                            }
995                        }
996
997                        debug_assert!(processed_fields < buffers.len());
998                        let buf = unsafe {
999                            // SAFETY: processed fields index can never exceed the projection indices.
1000                            buffers.get_unchecked_mut(processed_fields)
1001                        };
1002                        let mut add_null = false;
1003
1004                        // if we have null values argument, check if this field equal null value
1005                        if let Some(null_values) = null_values {
1006                            let field = if needs_escaping && !field.is_empty() {
1007                                unsafe { field.get_unchecked(1..field.len() - 1) }
1008                            } else {
1009                                field
1010                            };
1011
1012                            // SAFETY:
1013                            // process fields is in bounds
1014                            add_null = unsafe { null_values.is_null(field, idx as usize) }
1015                        }
1016                        if add_null {
1017                            buf.add_null(!parse_options.missing_is_null && field.is_empty())
1018                        } else {
1019                            buf.add(field, ignore_errors, needs_escaping, parse_options.missing_is_null)
1020                                .map_err(|e| {
1021                                    let bytes_offset = offset + field.as_ptr() as usize - start;
1022                                    let unparsable = String::from_utf8_lossy(field);
1023                                    let column_name = schema.get_at_index(idx as usize).unwrap().0;
1024                                    polars_err!(
1025                                        ComputeError:
1026                                        "could not parse `{}` as dtype `{}` at column '{}' (column number {})\n\n\
1027                                        The current offset in the file is {} bytes.\n\
1028                                        \n\
1029                                        You might want to try:\n\
1030                                        - increasing `infer_schema_length` (e.g. `infer_schema_length=10000`),\n\
1031                                        - specifying correct dtype with the `schema_overrides` argument\n\
1032                                        - setting `ignore_errors` to `True`,\n\
1033                                        - adding `{}` to the `null_values` list.\n\n\
1034                                        Original error: ```{}```",
1035                                        &unparsable,
1036                                        buf.dtype(),
1037                                        column_name,
1038                                        idx + 1,
1039                                        bytes_offset,
1040                                        &unparsable,
1041                                        e
1042                                    )
1043                                })?;
1044                        }
1045                        processed_fields += 1;
1046
1047                        // if we have all projected columns we are done with this line
1048                        match projection_iter.next() {
1049                            Some(p) => next_projected = p,
1050                            None => {
1051                                if bytes.get(read_sol - 1) == Some(&parse_options.eol_char) {
1052                                    bytes = &bytes[read_sol..];
1053                                } else {
1054                                    if !truncate_ragged_lines && read_sol < bytes.len() {
1055                                        polars_bail!(ComputeError: r#"found more fields than defined in 'Schema'
1056
1057Consider setting 'truncate_ragged_lines={}'."#, polars_error::constants::TRUE)
1058                                    }
1059                                    let bytes_rem = skip_this_line(
1060                                        unsafe { bytes.get_unchecked(read_sol - 1..) },
1061                                        parse_options.quote_char,
1062                                        parse_options.eol_char,
1063                                    );
1064                                    bytes = bytes_rem;
1065                                }
1066                                break;
1067                            },
1068                        }
1069                    }
1070                    idx += 1;
1071                },
1072            }
1073        }
1074
1075        // there can be lines that miss fields (also the comma values)
1076        // this means the splitter won't process them.
1077        // We traverse them to read them as null values.
1078        while processed_fields < projection.len() {
1079            debug_assert!(processed_fields < buffers.len());
1080            let buf = unsafe {
1081                // SAFETY: processed fields index can never exceed the projection indices.
1082                buffers.get_unchecked_mut(processed_fields)
1083            };
1084            buf.add_null(!parse_options.missing_is_null);
1085            processed_fields += 1;
1086        }
1087        line_count += 1;
1088    }
1089}
