76    /// Convert a categorical column to its local representation.
77    pub fn to_local(&self) -> Self {
78        let rev_map = self.get_rev_map();
79        let (physical_map, categories) = match rev_map.as_ref() {
80            RevMapping::Global(m, c, _) => (m, c),
81            RevMapping::Local(_, _) if !self.is_enum() => return self.clone(),
82            RevMapping::Local(_, _) => {
83                // Change dtype from Enum to Categorical
84                let mut local = self.clone();
85                local.physical.2 = Some(DataType::Categorical(
86                    Some(rev_map.clone()),
87                    self.get_ordering(),
88                ));
89                return local;
90            },
91        };
92
93        let local_rev_map = RevMapping::build_local(categories.clone());
94        // TODO: A fast path can possibly be implemented here:
95        // if all physical map keys are equal to their values,
96        // we can skip the apply and only update the rev_map
97        let local_ca = self
98            .physical()
99            .apply(|opt_v| opt_v.map(|v| *physical_map.get(&v).unwrap()));
100
101        let mut out = unsafe {
102            Self::from_cats_and_rev_map_unchecked(
103                local_ca,
104                local_rev_map.into(),
105                false,
106                self.get_ordering(),
107            )
108        };
109        out.set_fast_unique(self._can_fast_unique());
110
111        out
112    }

114    pub fn to_global(&self) -> PolarsResult<Self> {
115        polars_ensure!(using_string_cache(), string_cache_mismatch);
116        // Fast path
117        let categories = match &**self.get_rev_map() {
118            RevMapping::Global(_, _, _) => return Ok(self.clone()),
119            RevMapping::Local(categories, _) => categories,
120        };
121
122        // SAFETY: keys and values are in bounds
123        unsafe {
124            Ok(CategoricalChunked::from_keys_and_values_global(
125                self.name().clone(),
126                self.physical(),
127                self.len(),
128                categories,
129                self.get_ordering(),
130            ))
131        }
132    }

134    // Convert to fixed enum. Values not in categories are mapped to None.
135    pub fn to_enum(&self, categories: &Utf8ViewArray, hash: u128) -> Self {
136        // Fast paths
137        match self.get_rev_map().as_ref() {
138            RevMapping::Local(_, cur_hash) if hash == *cur_hash => {
139                return unsafe {
140                    CategoricalChunked::from_cats_and_rev_map_unchecked(
141                        self.physical().clone(),
142                        self.get_rev_map().clone(),
143                        true,
144                        self.get_ordering(),
145                    )
146                };
147            },
148            _ => (),
149        };
150        // Make a mapping from old idx to new idx
151        let old_rev_map = self.get_rev_map();
152
153        // Create map of old category -> idx for fast lookup.
154        let old_categories = old_rev_map.get_categories();
155        let old_idx_map: PlHashMap<&str, u32> = old_categories
156            .values_iter()
157            .zip(0..old_categories.len() as u32)
158            .collect();
159
160        #[allow(clippy::unnecessary_cast)]
161        let idx_map: PlHashMap<u32, u32> = categories
162            .values_iter()
163            .enumerate_idx()
164            .filter_map(|(new_idx, s)| old_idx_map.get(s).map(|old_idx| (*old_idx, new_idx as u32)))
165            .collect();
166
167        // Loop over the physicals and try get new idx
168        let new_phys: UInt32Chunked = self
169            .physical()
170            .into_iter()
171            .map(|opt_v: Option<u32>| opt_v.and_then(|v| idx_map.get(&v).copied()))
172            .collect();
173
174        // SAFETY: we created the physical from the enum categories
175        unsafe {
176            CategoricalChunked::from_cats_and_rev_map_unchecked(
177                new_phys,
178                Arc::new(RevMapping::Local(categories.clone(), hash)),
179                true,
180                self.get_ordering(),
181            )
182        }
183    }

214    /// Create a [`CategoricalChunked`] from a physical array and dtype.

219    pub unsafe fn from_cats_and_dtype_unchecked(idx: UInt32Chunked, dtype: DataType) -> Self {
220        debug_assert!(matches!(
221            dtype,
222            DataType::Enum { .. } | DataType::Categorical { .. }
223        ));
224        let mut logical = Logical::<UInt32Type, _>::new_logical::<CategoricalType>(idx);
225        logical.2 = Some(dtype);
226        Self {
227            physical: logical,
228            bit_settings: Default::default(),

232    /// Create a [`CategoricalChunked`] from an array of `idx` and an existing [`RevMapping`]:  `rev_map`.
233    ///
234    /// # Safety
235    /// Invariant in `v < rev_map.len() for v in idx` must hold.
236    pub unsafe fn from_cats_and_rev_map_unchecked(
237        idx: UInt32Chunked,
238        rev_map: Arc<RevMapping>,
239        is_enum: bool,
240        ordering: CategoricalOrdering,
241    ) -> Self {
242        let mut logical = Logical::<UInt32Type, _>::new_logical::<CategoricalType>(idx);
243        if is_enum {
244            logical.2 = Some(DataType::Enum(Some(rev_map), ordering));
245        } else {
246            logical.2 = Some(DataType::Categorical(Some(rev_map), ordering));
247        }
248        Self {
249            physical: logical,
250            bit_settings: Default::default(),
251        }
252    }

254    pub(crate) fn set_ordering(
255        mut self,
256        ordering: CategoricalOrdering,
257        keep_fast_unique: bool,
258    ) -> Self {
259        self.physical.2 = match self.dtype() {
260            DataType::Enum(_, _) => {
261                Some(DataType::Enum(Some(self.get_rev_map().clone()), ordering))
262            },
263            DataType::Categorical(_, _) => Some(DataType::Categorical(
264                Some(self.get_rev_map().clone()),
265                ordering,
266            )),
267            _ => panic!("implementation error"),
268        };
269
270        if !keep_fast_unique {
271            self.set_fast_unique(false)
272        }
273        self
274    }

292    /// True if all categories are represented in this array. When this is the case, the unique
293    /// values of the array are the categories.
294    pub fn _can_fast_unique(&self) -> bool {
295        self.bit_settings.contains(BitSettings::ORIGINAL)
296            && self.physical.chunks.len() == 1
297            && self.null_count() == 0
298    }

300    pub(crate) fn set_fast_unique(&mut self, toggle: bool) {
301        if toggle {
302            self.bit_settings.insert(BitSettings::ORIGINAL);
303        } else {
304            self.bit_settings.remove(BitSettings::ORIGINAL);
305        }
306    }

354    unsafe fn get_any_value_unchecked(&self, i: usize) -> AnyValue<'_> {
355        match self.physical.0.get_unchecked(i) {
356            Some(i) => match self.dtype() {
357                DataType::Enum(_, _) => AnyValue::Enum(i, self.get_rev_map(), SyncPtr::new_null()),
358                DataType::Categorical(_, _) => {
359                    AnyValue::Categorical(i, self.get_rev_map(), SyncPtr::new_null())
360                },
361                _ => unimplemented!(),
362            },
363            None => AnyValue::Null,
364        }

367    fn cast_with_options(&self, dtype: &DataType, options: CastOptions) -> PolarsResult<Series> {
368        match dtype {
369            DataType::String => {
370                let mapping = &**self.get_rev_map();
371
372                let mut builder =
373                    StringChunkedBuilder::new(self.physical.name().clone(), self.len());
374
375                let f = |idx: u32| mapping.get(idx);
376
377                if !self.physical.has_nulls() {
378                    self.physical
379                        .into_no_null_iter()
380                        .for_each(|idx| builder.append_value(f(idx)));
381                } else {
382                    self.physical.into_iter().for_each(|opt_idx| {
383                        builder.append_option(opt_idx.map(f));
384                    });
385                }
386
387                let ca = builder.finish();
388                Ok(ca.into_series())
389            },
390            DataType::UInt32 => {
391                let ca = unsafe {
392                    UInt32Chunked::from_chunks(
393                        self.physical.name().clone(),
394                        self.physical.chunks.clone(),
395                    )
396                };
397                Ok(ca.into_series())
398            },
399            #[cfg(feature = "dtype-categorical")]
400            DataType::Enum(Some(rev_map), ordering) => {
401                let RevMapping::Local(categories, hash) = &**rev_map else {
402                    polars_bail!(ComputeError: "can not cast to enum with global mapping")
403                };
404                Ok(self
405                    .to_enum(categories, *hash)
406                    .set_ordering(*ordering, true)
407                    .into_series()
408                    .with_name(self.name().clone()))
409            },
410            DataType::Enum(None, _) => {
411                polars_bail!(ComputeError: "can not cast to enum without categories present")
412            },
413            #[cfg(feature = "dtype-categorical")]
414            DataType::Categorical(rev_map, ordering) => {
415                // Casting from an Enum to a local or global
416                if matches!(self.dtype(), DataType::Enum(_, _)) && rev_map.is_none() {
417                    if using_string_cache() {
418                        return Ok(self
419                            .to_global()?
420                            .set_ordering(*ordering, true)
421                            .into_series());
422                    } else {
423                        return Ok(self.to_local().set_ordering(*ordering, true).into_series());
424                    }
425                }
426                // If casting to lexical categorical, set sorted flag as not set
427
428                let mut ca = self.clone().set_ordering(*ordering, true);
429                if ca.uses_lexical_ordering() {
430                    ca.physical.set_sorted_flag(IsSorted::Not);
431                }
432                Ok(ca.into_series())
433            },
434            dt if dt.is_primitive_numeric() => {
435                // Apply the cast to the categories and then index into the casted series.
436                // This has to be local for the gather.
437                let slf = self.to_local();
438                let categories = StringChunked::with_chunk(
439                    slf.physical.name().clone(),
440                    slf.get_rev_map().get_categories().clone(),
441                );
442                let casted_series = categories.cast_with_options(dtype, options)?;
443
444                #[cfg(feature = "bigidx")]
445                {
446                    let s = slf.physical.cast_with_options(&DataType::UInt64, options)?;
447                    Ok(unsafe { casted_series.take_unchecked(s.u64()?) })
448                }
449                #[cfg(not(feature = "bigidx"))]
450                {
451                    // SAFETY: Invariant of categorical means indices are in bound
452                    Ok(unsafe { casted_series.take_unchecked(&slf.physical) })
453                }
454            },
455            _ => self.physical.cast_with_options(dtype, options),
456        }
457    }
458}

460pub struct CatIter<'a> {
461    rev: &'a RevMapping,
462    iter: Box<dyn PolarsIterator<Item = Option<u32>> + 'a>,
463}
464
465unsafe impl TrustedLen for CatIter<'_> {}
466
467impl<'a> Iterator for CatIter<'a> {
468    type Item = Option<&'a str>;
469
470    fn next(&mut self) -> Option<Self::Item> {
471        self.iter.next().map(|item| {
472            item.map(|idx| {
473                // SAFETY:
474                // all categories are in bound
475                unsafe { self.rev.get_unchecked(idx) }
476            })
477        })
478    }
479
480    fn size_hint(&self) -> (usize, Option<usize>) {
481        self.iter.size_hint()
482    }
483}
484
485impl DoubleEndedIterator for CatIter<'_> {
486    fn next_back(&mut self) -> Option<Self::Item> {
487        self.iter.next_back().map(|item| {
488            item.map(|idx| {
489                // SAFETY:
490                // all categories are in bound
491                unsafe { self.rev.get_unchecked(idx) }
492            })
493        })
494    }
495}

504    #[test]
505    fn test_categorical_round_trip() -> PolarsResult<()> {
506        let _lock = SINGLE_LOCK.lock();
507        disable_string_cache();
508        let slice = &[
509            Some("foo"),
510            None,
511            Some("bar"),
512            Some("foo"),
513            Some("foo"),
514            Some("bar"),
515        ];
516        let ca = StringChunked::new(PlSmallStr::from_static("a"), slice);
517        let ca = ca.cast(&DataType::Categorical(None, Default::default()))?;
518        let ca = ca.categorical().unwrap();
519
520        let arr = ca.to_arrow(CompatLevel::newest(), false);
521        let s = Series::try_from((PlSmallStr::from_static("foo"), arr))?;
522        assert!(matches!(s.dtype(), &DataType::Categorical(_, _)));
523        assert_eq!(s.null_count(), 1);
524        assert_eq!(s.len(), 6);

563    #[test]
564    fn test_categorical_flow() -> PolarsResult<()> {
565        let _lock = SINGLE_LOCK.lock();
566        disable_string_cache();
567
568        // tests several things that may lose the dtype information
569        let s = Series::new(PlSmallStr::from_static("a"), vec!["a", "b", "c"])
570            .cast(&DataType::Categorical(None, Default::default()))?;
571
572        assert_eq!(

574            Field::new(
575                PlSmallStr::from_static("a"),
576                DataType::Categorical(None, Default::default())

578        );
579        assert!(matches!(
580            s.get(0)?,
581            AnyValue::Categorical(0, RevMapping::Local(_, _), _)
582        ));
583
584        let groups = s.group_tuples(false, true);
585        let aggregated = unsafe { s.agg_list(&groups?) };
586        match aggregated.get(0)? {
587            AnyValue::List(s) => {
588                assert!(matches!(s.dtype(), DataType::Categorical(_, _)));
589                let str_s = s.cast(&DataType::String).unwrap();
590                assert_eq!(str_s.get(0)?, AnyValue::String("a"));
591                assert_eq!(s.len(), 1);
592            },
593            _ => panic!(),
594        }
595        let flat = aggregated.explode()?;
596        let ca = flat.categorical().unwrap();
597        let vals = ca.iter_str().map(|v| v.unwrap()).collect::<Vec<_>>();
598        assert_eq!(vals, &["a", "b", "c"]);
