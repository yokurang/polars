454    /// Extract a numerical value from the AnyValue
455    #[doc(hidden)]
456    #[inline]
457    pub fn extract<T: NumCast>(&self) -> Option<T> {
458        use AnyValue::*;
459        match self {
460            Int8(v) => NumCast::from(*v),
461            Int16(v) => NumCast::from(*v),
462            Int32(v) => NumCast::from(*v),
463            Int64(v) => NumCast::from(*v),
464            Int128(v) => NumCast::from(*v),
465            UInt8(v) => NumCast::from(*v),
466            UInt16(v) => NumCast::from(*v),
467            UInt32(v) => NumCast::from(*v),
468            UInt64(v) => NumCast::from(*v),
469            Float32(v) => NumCast::from(*v),
470            Float64(v) => NumCast::from(*v),
471            #[cfg(feature = "dtype-date")]
472            Date(v) => NumCast::from(*v),
473            #[cfg(feature = "dtype-datetime")]
474            Datetime(v, _, _) | DatetimeOwned(v, _, _) => NumCast::from(*v),
475            #[cfg(feature = "dtype-time")]
476            Time(v) => NumCast::from(*v),
477            #[cfg(feature = "dtype-duration")]
478            Duration(v, _) => NumCast::from(*v),
479            #[cfg(feature = "dtype-decimal")]
480            Decimal(v, scale) => {
481                if *scale == 0 {
482                    NumCast::from(*v)
483                } else {
484                    let f: Option<f64> = NumCast::from(*v);
485                    NumCast::from(f? / 10f64.powi(*scale as _))
486                }
487            },
488            Boolean(v) => NumCast::from(if *v { 1 } else { 0 }),
489            String(v) => {
490                if let Ok(val) = (*v).parse::<i128>() {
491                    NumCast::from(val)
492                } else {
493                    NumCast::from((*v).parse::<f64>().ok()?)
494                }
495            },
496            StringOwned(v) => String(v.as_str()).extract(),
497            _ => None,
498        }
499    }

569    /// Cast `AnyValue` to the provided data type and return a new `AnyValue` with type `dtype`,
570    /// if possible.
571    pub fn strict_cast(&self, dtype: &'a DataType) -> Option<AnyValue<'a>> {
572        let new_av = match (self, dtype) {
573            // to numeric
574            (av, DataType::UInt8) => AnyValue::UInt8(av.extract::<u8>()?),
575            (av, DataType::UInt16) => AnyValue::UInt16(av.extract::<u16>()?),
576            (av, DataType::UInt32) => AnyValue::UInt32(av.extract::<u32>()?),
577            (av, DataType::UInt64) => AnyValue::UInt64(av.extract::<u64>()?),
578            (av, DataType::Int8) => AnyValue::Int8(av.extract::<i8>()?),
579            (av, DataType::Int16) => AnyValue::Int16(av.extract::<i16>()?),
580            (av, DataType::Int32) => AnyValue::Int32(av.extract::<i32>()?),
581            (av, DataType::Int64) => AnyValue::Int64(av.extract::<i64>()?),
582            (av, DataType::Int128) => AnyValue::Int128(av.extract::<i128>()?),
583            (av, DataType::Float32) => AnyValue::Float32(av.extract::<f32>()?),
584            (av, DataType::Float64) => AnyValue::Float64(av.extract::<f64>()?),
585
586            // to boolean
587            (AnyValue::UInt8(v), DataType::Boolean) => AnyValue::Boolean(*v != u8::default()),
588            (AnyValue::UInt16(v), DataType::Boolean) => AnyValue::Boolean(*v != u16::default()),
589            (AnyValue::UInt32(v), DataType::Boolean) => AnyValue::Boolean(*v != u32::default()),
590            (AnyValue::UInt64(v), DataType::Boolean) => AnyValue::Boolean(*v != u64::default()),
591            (AnyValue::Int8(v), DataType::Boolean) => AnyValue::Boolean(*v != i8::default()),
592            (AnyValue::Int16(v), DataType::Boolean) => AnyValue::Boolean(*v != i16::default()),
593            (AnyValue::Int32(v), DataType::Boolean) => AnyValue::Boolean(*v != i32::default()),
594            (AnyValue::Int64(v), DataType::Boolean) => AnyValue::Boolean(*v != i64::default()),
595            (AnyValue::Int128(v), DataType::Boolean) => AnyValue::Boolean(*v != i128::default()),
596            (AnyValue::Float32(v), DataType::Boolean) => AnyValue::Boolean(*v != f32::default()),
597            (AnyValue::Float64(v), DataType::Boolean) => AnyValue::Boolean(*v != f64::default()),
598
599            // to string
600            (AnyValue::String(v), DataType::String) => AnyValue::String(v),
601            (AnyValue::StringOwned(v), DataType::String) => AnyValue::StringOwned(v.clone()),
602
603            (av, DataType::String) => {
604                let mut tmp = vec![];
605                if av.is_unsigned_integer() {
606                    let val = av.extract::<u64>()?;
607                    SerPrimitive::write(&mut tmp, val);
608                } else if av.is_float() {
609                    let val = av.extract::<f64>()?;
610                    SerPrimitive::write(&mut tmp, val);
611                } else {
612                    let val = av.extract::<i64>()?;
613                    SerPrimitive::write(&mut tmp, val);
614                }
615                AnyValue::StringOwned(PlSmallStr::from_str(std::str::from_utf8(&tmp).unwrap()))
616            },
617
618            // to binary
619            (AnyValue::String(v), DataType::Binary) => AnyValue::Binary(v.as_bytes()),
620
621            // to datetime
622            #[cfg(feature = "dtype-datetime")]
623            (av, DataType::Datetime(tu, tz)) if av.is_primitive_numeric() => {
624                AnyValue::Datetime(av.extract::<i64>()?, *tu, tz.as_ref())
625            },
626            #[cfg(all(feature = "dtype-datetime", feature = "dtype-date"))]
627            (AnyValue::Date(v), DataType::Datetime(tu, _)) => AnyValue::Datetime(
628                match tu {
629                    TimeUnit::Nanoseconds => (*v as i64) * NS_IN_DAY,
630                    TimeUnit::Microseconds => (*v as i64) * US_IN_DAY,
631                    TimeUnit::Milliseconds => (*v as i64) * MS_IN_DAY,
632                },
633                *tu,
634                None,
635            ),
636            #[cfg(feature = "dtype-datetime")]
637            (
638                AnyValue::Datetime(v, tu, _) | AnyValue::DatetimeOwned(v, tu, _),
639                DataType::Datetime(tu_r, tz_r),
640            ) => AnyValue::Datetime(
641                match (tu, tu_r) {
642                    (TimeUnit::Nanoseconds, TimeUnit::Microseconds) => *v / 1_000i64,
643                    (TimeUnit::Nanoseconds, TimeUnit::Milliseconds) => *v / 1_000_000i64,
644                    (TimeUnit::Microseconds, TimeUnit::Nanoseconds) => *v * 1_000i64,
645                    (TimeUnit::Microseconds, TimeUnit::Milliseconds) => *v / 1_000i64,
646                    (TimeUnit::Milliseconds, TimeUnit::Microseconds) => *v * 1_000i64,
647                    (TimeUnit::Milliseconds, TimeUnit::Nanoseconds) => *v * 1_000_000i64,
648                    _ => *v,
649                },
650                *tu_r,
651                tz_r.as_ref(),
652            ),
653
654            // to date
655            #[cfg(feature = "dtype-date")]
656            (av, DataType::Date) if av.is_primitive_numeric() => {
657                AnyValue::Date(av.extract::<i32>()?)
658            },
659            #[cfg(all(feature = "dtype-date", feature = "dtype-datetime"))]
660            (AnyValue::Datetime(v, tu, _) | AnyValue::DatetimeOwned(v, tu, _), DataType::Date) => {
661                AnyValue::Date(match tu {
662                    TimeUnit::Nanoseconds => *v / NS_IN_DAY,
663                    TimeUnit::Microseconds => *v / US_IN_DAY,
664                    TimeUnit::Milliseconds => *v / MS_IN_DAY,
665                } as i32)
666            },
667
668            // to time
669            #[cfg(feature = "dtype-time")]
670            (av, DataType::Time) if av.is_primitive_numeric() => {
671                AnyValue::Time(av.extract::<i64>()?)
672            },
673            #[cfg(all(feature = "dtype-time", feature = "dtype-datetime"))]
674            (AnyValue::Datetime(v, tu, _) | AnyValue::DatetimeOwned(v, tu, _), DataType::Time) => {
675                AnyValue::Time(match tu {
676                    TimeUnit::Nanoseconds => *v % NS_IN_DAY,
677                    TimeUnit::Microseconds => (*v % US_IN_DAY) * 1_000i64,
678                    TimeUnit::Milliseconds => (*v % MS_IN_DAY) * 1_000_000i64,
679                })
680            },
681
682            // to duration
683            #[cfg(feature = "dtype-duration")]
684            (av, DataType::Duration(tu)) if av.is_primitive_numeric() => {
685                AnyValue::Duration(av.extract::<i64>()?, *tu)
686            },
687            #[cfg(all(feature = "dtype-duration", feature = "dtype-time"))]
688            (AnyValue::Time(v), DataType::Duration(tu)) => AnyValue::Duration(
689                match *tu {
690                    TimeUnit::Nanoseconds => *v,
691                    TimeUnit::Microseconds => *v / 1_000i64,
692                    TimeUnit::Milliseconds => *v / 1_000_000i64,
693                },
694                *tu,
695            ),
696            #[cfg(feature = "dtype-duration")]
697            (AnyValue::Duration(v, tu), DataType::Duration(tu_r)) => AnyValue::Duration(
698                match (tu, tu_r) {
699                    (_, _) if tu == tu_r => *v,
700                    (TimeUnit::Nanoseconds, TimeUnit::Microseconds) => *v / 1_000i64,
701                    (TimeUnit::Nanoseconds, TimeUnit::Milliseconds) => *v / 1_000_000i64,
702                    (TimeUnit::Microseconds, TimeUnit::Nanoseconds) => *v * 1_000i64,
703                    (TimeUnit::Microseconds, TimeUnit::Milliseconds) => *v / 1_000i64,
704                    (TimeUnit::Milliseconds, TimeUnit::Microseconds) => *v * 1_000i64,
705                    (TimeUnit::Milliseconds, TimeUnit::Nanoseconds) => *v * 1_000_000i64,
706                    _ => *v,
707                },
708                *tu_r,
709            ),
710
711            // to decimal
712            #[cfg(feature = "dtype-decimal")]
713            (av, DataType::Decimal(prec, scale)) if av.is_integer() => {
714                let value = av.try_extract::<i128>().unwrap();
715                let scale = scale.unwrap_or(0);
716                let factor = 10_i128.pow(scale as _); // Conversion to u32 is safe, max value is 38.
717                let converted = value.checked_mul(factor)?;
718
719                // Check if the converted value fits into the specified precision
720                let prec = prec.unwrap_or(38) as u32;
721                let num_digits = (converted.abs() as f64).log10().ceil() as u32;
722                if num_digits > prec {
723                    return None;
724                }
725
726                AnyValue::Decimal(converted, scale)
727            },
728            #[cfg(feature = "dtype-decimal")]
729            (AnyValue::Decimal(value, scale_av), DataType::Decimal(_, scale)) => {
730                let Some(scale) = scale else {
731                    return Some(self.clone());
732                };
733                // TODO: Allow lossy conversion?
734                let scale_diff = scale.checked_sub(*scale_av)?;
735                let factor = 10_i128.pow(scale_diff as _); // Conversion is safe, max value is 38.
736                let converted = value.checked_mul(factor)?;
737                AnyValue::Decimal(converted, *scale)
738            },
739
740            // to self
741            (av, dtype) if av.dtype() == *dtype => self.clone(),
742
743            _ => return None,
744        };
745        Some(new_av)
746    }

812    pub fn hash_impl<H: Hasher>(&self, state: &mut H, cheap: bool) {
813        use AnyValue::*;
814        std::mem::discriminant(self).hash(state);
815        match self {
816            Int8(v) => v.hash(state),
817            Int16(v) => v.hash(state),
818            Int32(v) => v.hash(state),
819            Int64(v) => v.hash(state),
820            Int128(v) => feature_gated!("dtype-i128", v.hash(state)),
821            UInt8(v) => v.hash(state),
822            UInt16(v) => v.hash(state),
823            UInt32(v) => v.hash(state),
824            UInt64(v) => v.hash(state),
825            String(v) => v.hash(state),
826            StringOwned(v) => v.hash(state),
827            Float32(v) => v.to_ne_bytes().hash(state),
828            Float64(v) => v.to_ne_bytes().hash(state),
829            Binary(v) => v.hash(state),
830            BinaryOwned(v) => v.hash(state),
831            Boolean(v) => v.hash(state),
832            List(v) => {
833                if !cheap {
834                    Hash::hash(&Wrap(v.clone()), state)
835                }
836            },
837            #[cfg(feature = "dtype-array")]
838            Array(v, width) => {
839                if !cheap {
840                    Hash::hash(&Wrap(v.clone()), state)
841                }
842                width.hash(state)
843            },
844            #[cfg(feature = "dtype-date")]
845            Date(v) => v.hash(state),
846            #[cfg(feature = "dtype-datetime")]
847            Datetime(v, tu, tz) => {
848                v.hash(state);
849                tu.hash(state);
850                tz.hash(state);
851            },
852            #[cfg(feature = "dtype-datetime")]
853            DatetimeOwned(v, tu, tz) => {
854                v.hash(state);
855                tu.hash(state);
856                tz.hash(state);
857            },
858            #[cfg(feature = "dtype-duration")]
859            Duration(v, tz) => {
860                v.hash(state);
861                tz.hash(state);
862            },
863            #[cfg(feature = "dtype-time")]
864            Time(v) => v.hash(state),
865            #[cfg(feature = "dtype-categorical")]
866            Categorical(v, _, _)
867            | CategoricalOwned(v, _, _)
868            | Enum(v, _, _)
869            | EnumOwned(v, _, _) => v.hash(state),
870            #[cfg(feature = "object")]
871            Object(_) => {},
872            #[cfg(feature = "object")]
873            ObjectOwned(_) => {},
874            #[cfg(feature = "dtype-struct")]
875            Struct(_, _, _) => {
876                if !cheap {
877                    let mut buf = vec![];
878                    self._materialize_struct_av(&mut buf);
879                    buf.hash(state)
880                }
881            },
882            #[cfg(feature = "dtype-struct")]
883            StructOwned(v) => v.0.hash(state),
884            #[cfg(feature = "dtype-decimal")]
885            Decimal(v, k) => {
886                v.hash(state);
887                k.hash(state);
888            },
889            Null => {},
890        }
891    }

1135impl AnyValue<'_> {
1136    #[inline]
1137    pub fn eq_missing(&self, other: &Self, null_equal: bool) -> bool {
1138        fn struct_owned_value_iter<'a>(
1139            v: &'a (Vec<AnyValue<'_>>, Vec<Field>),
1140        ) -> impl ExactSizeIterator<Item = AnyValue<'a>> {
1141            v.0.iter().map(|v| v.as_borrowed())
1142        }
1143        fn struct_value_iter(
1144            idx: usize,
1145            arr: &StructArray,
1146        ) -> impl ExactSizeIterator<Item = AnyValue<'_>> {
1147            assert!(idx < arr.len());
1148
1149            arr.values().iter().map(move |field_arr| unsafe {
1150                // SAFETY: We asserted before that idx is smaller than the array length. Since it
1151                // is an invariant of StructArray that all fields have the same length this is fine
1152                // to do.
1153                field_arr.get_unchecked(idx)
1154            })
1155        }
1156
1157        fn struct_eq_missing<'a>(
1158            l: impl ExactSizeIterator<Item = AnyValue<'a>>,
1159            r: impl ExactSizeIterator<Item = AnyValue<'a>>,
1160            null_equal: bool,
1161        ) -> bool {
1162            if l.len() != r.len() {
1163                return false;
1164            }
1165
1166            l.zip(r).all(|(lv, rv)| lv.eq_missing(&rv, null_equal))
1167        }
1168
1169        use AnyValue::*;
1170        match (self, other) {
1171            // Map to borrowed.
1172            (StringOwned(l), r) => AnyValue::String(l.as_str()) == *r,
1173            (BinaryOwned(l), r) => AnyValue::Binary(l.as_slice()) == *r,
1174            #[cfg(feature = "object")]
1175            (ObjectOwned(l), r) => AnyValue::Object(&*l.0) == *r,
1176            (l, StringOwned(r)) => *l == AnyValue::String(r.as_str()),
1177            (l, BinaryOwned(r)) => *l == AnyValue::Binary(r.as_slice()),
1178            #[cfg(feature = "object")]
1179            (l, ObjectOwned(r)) => *l == AnyValue::Object(&*r.0),
1180            #[cfg(feature = "dtype-datetime")]
1181            (DatetimeOwned(lv, ltu, ltz), r) => {
1182                Datetime(*lv, *ltu, ltz.as_ref().map(|v| v.as_ref())) == *r
1183            },
1184            #[cfg(feature = "dtype-datetime")]
1185            (l, DatetimeOwned(rv, rtu, rtz)) => {
1186                *l == Datetime(*rv, *rtu, rtz.as_ref().map(|v| v.as_ref()))
1187            },
1188            #[cfg(feature = "dtype-categorical")]
1189            (CategoricalOwned(lv, lrev, larr), r) => Categorical(*lv, lrev.as_ref(), *larr) == *r,
1190            #[cfg(feature = "dtype-categorical")]
1191            (l, CategoricalOwned(rv, rrev, rarr)) => *l == Categorical(*rv, rrev.as_ref(), *rarr),
1192            #[cfg(feature = "dtype-categorical")]
1193            (EnumOwned(lv, lrev, larr), r) => Enum(*lv, lrev.as_ref(), *larr) == *r,
1194            #[cfg(feature = "dtype-categorical")]
1195            (l, EnumOwned(rv, rrev, rarr)) => *l == Enum(*rv, rrev.as_ref(), *rarr),
1196
1197            // Comparison with null.
1198            (Null, Null) => null_equal,
1199            (Null, _) => false,
1200            (_, Null) => false,
1201
1202            // Equality between equal types.
1203            (Boolean(l), Boolean(r)) => *l == *r,
1204            (UInt8(l), UInt8(r)) => *l == *r,
1205            (UInt16(l), UInt16(r)) => *l == *r,
1206            (UInt32(l), UInt32(r)) => *l == *r,
1207            (UInt64(l), UInt64(r)) => *l == *r,
1208            (Int8(l), Int8(r)) => *l == *r,
1209            (Int16(l), Int16(r)) => *l == *r,
1210            (Int32(l), Int32(r)) => *l == *r,
1211            (Int64(l), Int64(r)) => *l == *r,
1212            (Int128(l), Int128(r)) => *l == *r,
1213            (Float32(l), Float32(r)) => l.to_total_ord() == r.to_total_ord(),
1214            (Float64(l), Float64(r)) => l.to_total_ord() == r.to_total_ord(),
1215            (String(l), String(r)) => l == r,
1216            (Binary(l), Binary(r)) => l == r,
1217            #[cfg(feature = "dtype-time")]
1218            (Time(l), Time(r)) => *l == *r,
1219            #[cfg(all(feature = "dtype-datetime", feature = "dtype-date"))]
1220            (Date(l), Date(r)) => *l == *r,
1221            #[cfg(all(feature = "dtype-datetime", feature = "dtype-date"))]
1222            (Datetime(l, tul, tzl), Datetime(r, tur, tzr)) => {
1223                *l == *r && *tul == *tur && tzl == tzr
1224            },
1225            (List(l), List(r)) => l == r,
1226            #[cfg(feature = "dtype-categorical")]
1227            (Categorical(idx_l, rev_l, ptr_l), Categorical(idx_r, rev_r, ptr_r)) => {
1228                if !same_revmap(rev_l, *ptr_l, rev_r, *ptr_r) {
1229                    // We can't support this because our Hash impl directly hashes the index. If you
1230                    // add support for this we must change the Hash impl.
1231                    unimplemented!(
1232                        "comparing categoricals with different revmaps is not supported"
1233                    );
1234                }
1235
1236                idx_l == idx_r
1237            },
1238            #[cfg(feature = "dtype-categorical")]
1239            (Enum(idx_l, rev_l, ptr_l), Enum(idx_r, rev_r, ptr_r)) => {
1240                // We can't support this because our Hash impl directly hashes the index. If you
1241                // add support for this we must change the Hash impl.
1242                if !same_revmap(rev_l, *ptr_l, rev_r, *ptr_r) {
1243                    unimplemented!("comparing enums with different revmaps is not supported");
1244                }
1245
1246                idx_l == idx_r
1247            },
1248            #[cfg(feature = "dtype-duration")]
1249            (Duration(l, tu_l), Duration(r, tu_r)) => l == r && tu_l == tu_r,
1250
1251            #[cfg(feature = "dtype-struct")]
1252            (StructOwned(l), StructOwned(r)) => struct_eq_missing(
1253                struct_owned_value_iter(l.as_ref()),
1254                struct_owned_value_iter(r.as_ref()),
1255                null_equal,
1256            ),
1257            #[cfg(feature = "dtype-struct")]
1258            (StructOwned(l), Struct(idx, arr, _)) => struct_eq_missing(
1259                struct_owned_value_iter(l.as_ref()),
1260                struct_value_iter(*idx, arr),
1261                null_equal,
1262            ),
1263            #[cfg(feature = "dtype-struct")]
1264            (Struct(idx, arr, _), StructOwned(r)) => struct_eq_missing(
1265                struct_value_iter(*idx, arr),
1266                struct_owned_value_iter(r.as_ref()),
1267                null_equal,
1268            ),
1269            #[cfg(feature = "dtype-struct")]
1270            (Struct(l_idx, l_arr, _), Struct(r_idx, r_arr, _)) => struct_eq_missing(
1271                struct_value_iter(*l_idx, l_arr),
1272                struct_value_iter(*r_idx, r_arr),
1273                null_equal,
1274            ),
1275            #[cfg(feature = "dtype-decimal")]
1276            (Decimal(l_v, l_s), Decimal(r_v, r_s)) => {
1277                // l_v / 10**l_s == r_v / 10**r_s
1278                if l_s == r_s && l_v == r_v || *l_v == 0 && *r_v == 0 {
1279                    true
1280                } else if l_s < r_s {
1281                    // l_v * 10**(r_s - l_s) == r_v
1282                    if let Some(lhs) = (|| {
1283                        let exp = i128::checked_pow(10, (r_s - l_s).try_into().ok()?)?;
1284                        l_v.checked_mul(exp)
1285                    })() {
1286                        lhs == *r_v
1287                    } else {
1288                        false
1289                    }
1290                } else {
1291                    // l_v == r_v * 10**(l_s - r_s)
1292                    if let Some(rhs) = (|| {
1293                        let exp = i128::checked_pow(10, (l_s - r_s).try_into().ok()?)?;
1294                        r_v.checked_mul(exp)
1295                    })() {
1296                        *l_v == rhs
1297                    } else {
1298                        false
1299                    }
1300                }
1301            },
1302            #[cfg(feature = "object")]
1303            (Object(l), Object(r)) => l == r,
1304            #[cfg(feature = "dtype-array")]
1305            (Array(l_values, l_size), Array(r_values, r_size)) => {
1306                if l_size != r_size {
1307                    return false;
1308                }
1309
1310                debug_assert_eq!(l_values.len(), *l_size);
1311                debug_assert_eq!(r_values.len(), *r_size);
1312
1313                let mut is_equal = true;
1314                for i in 0..*l_size {
1315                    let l = unsafe { l_values.get_unchecked(i) };
1316                    let r = unsafe { r_values.get_unchecked(i) };
1317
1318                    is_equal &= l.eq_missing(&r, null_equal);
1319                }
1320                is_equal
1321            },
1322
1323            (l, r) if l.to_i128().is_some() && r.to_i128().is_some() => l.to_i128() == r.to_i128(),
1324            (l, r) if l.to_f64().is_some() && r.to_f64().is_some() => {
1325                l.to_f64().unwrap().to_total_ord() == r.to_f64().unwrap().to_total_ord()
1326            },
1327
1328            (_, _) => {
1329                unimplemented!(
1330                    "scalar eq_missing for mixed dtypes {self:?} and {other:?} is not supported"
1331                )
1332            },
1333        }
1334    }
